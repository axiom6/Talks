Supporting Methods

  

  text:( str ) ->
    if str? then str else ""

  atts:( attribs ) ->
    return "" if not attribs?
    html = ""
    for key, val of attribs
      html += """ #{key}="#{val}#{'"'}"""
    html

  indent:(n) ->
    str = "\n"
    for i in [0...n]
      str += "  "     # Indent 2 spaces
    str
